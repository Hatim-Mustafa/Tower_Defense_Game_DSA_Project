@startuml

participant "GameManager" as GM
participant "wavePattern[]" as WP
participant "Enemy Factory" as EF
participant "enemies[]" as EL

GM -> GM : startNextWave()
activate GM
GM -> WP : wavePattern.clear() ; fill based on currentWave
note right of WP : Wave 1 → 7 Red\nWave 2 → 7R+5B\nWave 3 → 7R+5B+5G\nWave 4-5 → 7R+5B+10G
GM -> GM : maxEnemies = wavePattern.size()\n         enemiesSpawned = 0\n         waveActive = true
deactivate GM

loop [waveActive && spawnTimer >= spawnDelay && enemiesSpawned < maxEnemies]
  GM -> WP  : type = wavePattern[enemiesSpawned]
  alt type == 0
    GM -> EF : new RedEnemy(pathHead, smart=true)
  else type == 1
    GM -> EF : new BlueEnemy(pathHead, smart=true)
  else type == 2
    GM -> EF : new GreenEnemy(pathHead, smart=true)
  end
  EF --> EL : push_back(enemy)
  GM -> GM  : updateSmartEnemyPaths()
  GM -> GM  : enemiesSpawned++
end

Combat Round

SQ-04 
@startuml
skinparam sequenceArrowColor #6080c0
skinparam sequenceLifeLineBorderColor #3050a0
skinparam participantBackgroundColor #202840
skinparam participantBorderColor #3050a0
skinparam participantFontColor #d0e0ff

participant "GameManager" as GM
participant "Tower" as T
participant "Projectile" as Pr
participant "Enemy" as E
participant "Player (obj)" as P

loop [each Tower t in towers[]]
  GM -> T  : t.update(dt, enemies, player)
  activate T
  T -> T   : timer += dt\n[timer >= 1/fireRate?]
  T -> T   : findTarget(enemies)\n(First / Strongest / Fastest mode)
  T -> Pr  : new Projectile(pos, target, damage, speed)
  T --> T  : projectiles.push_back(proj)
  deactivate T
  loop [each Projectile p in tower.projectiles[]]
    GM -> Pr : p.update(dt, player)
    activate Pr
    Pr -> Pr : move toward target
    Pr -> Pr : distance to target < hitRadius?
    Pr -> E  : takeDamage(damage, player)
    activate E
    alt enemy.health <= 0
      E -> P   : player.increaseMoney(reward)
      E -> E   : alive = false
    end
    deactivate E
    deactivate Pr
  end
end
@enduml

GM -> E  : reachedEndOfPath() && isAlive()?
E --> GM : true
GM -> RE : reachedEnd.push_back(enemy)
GM -> GM : enemies.erase(enemy)
loop [each e in reachedEnd]
  GM -> P  : player.updateHealth(e)
  note right of P : health -= e.health\n(or fixed penalty)
  GM -> E  : delete e
end
GM -> P  : player.getHealth() <= 0 ?
alt yes
  GM -> GM : isGameover = true
  note right of GM : stops all further\ngame updates
end

AC-02 
@startuml
skinparam activityBackgroundColor #202840
skinparam activityBorderColor #3050a0
skinparam activityFontColor #d0e0ff
skinparam arrowColor #6080c0

start
:startNextWave()\n  build wavePattern[]\n  waveActive = true;
while (waveActive?) is (yes)
  if (spawnTimer >= spawnDelay\n&& enemiesSpawned < maxEnemies?) then (yes)
    :Spawn next enemy\n(Red / Blue / Green);
    :updateSmartEnemyPaths();
    :enemiesSpawned++;
  else (no)
    :Wait / move enemies\n& update towers;
  endif
  if (player.health <= 0?) then (yes)
    :isGameover = true;
    stop
  endif
  if (enemiesSpawned == maxEnemies\n&& enemies.empty()?) then (yes)
    :waveActive = false;
  endif
endwhile (no)
if (currentWave > 5?) then (yes)
  :isGameComplete() - YOU WIN!;
  stop
else (no)
  :wavePauseTimer counts up\nstartNextWave() again;
endif
stop
@enduml

@startuml
skinparam activityBackgroundColor #202840

start
fork
  :PLACEMENT FLOW;
  :Player clicks Shop icon;
  :isDragging = true;
  :Drag tower sprite to cell;
  :Mouse released;
  if (cell buildable\n&& player.money > cost?) then (yes)
    :decreaseMoney(TowerCost);
    :new Tower(pos) → towers[];
    :updateSmartEnemyPaths();
  else (no)
    :Snap back — no placement;
  endif
fork again
  :UPGRADE FLOW;
  :Player clicks a Tower;
  :tower.isSelected = true;
  if (Click "Upgrade" button?) then (yes)
    :showUpgradeOptions = true;
    :Display up to 2 child nodes\nof UpgradeNode tree;
    :Player clicks an upgrade;
    if (player.money >= upgrade.UpgradeCost?) then (yes)
      :Apply damage/range/fireRate/speed boost;
      :decreaseMoney(UpgradeCost);
      :level++;
      :updateSprite();
    else (no)
      :Not enough money — no upgrade;
    endif
  else if (Click "Mode" button?) then (yes)
    :showModeOptions = true;
    :Choose: First / Strongest / Fastest;
    :targetingMode = chosen;
  endif
end fork
stop
